\newcommand{\outlinestart}[1]
{
\ifthenelse{\boolean{#1}}{\begin{enumerate}}{}
}

\newcommand{\outline}[2]
{
\ifthenelse{\boolean{#1}}{\it \color {red} \item #2 \rm \color {black}}{}
}

\newcommand{\outlineend}[1]
{
\ifthenelse{\boolean{#1}}{\end{enumerate}}{}
}
